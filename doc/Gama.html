def notify(new_user)
  self.users.each do |user|
    if user.id != new_user.id
      SocketController.user_joined(user, new_user)
    end
  end

def surrender(loser)
  gama_id = loser.gama_id
  self.users.each do |user|
    user.leave_game
  end
  self.done = true
  self.save
  SocketController.gameover(gama_id)
end
